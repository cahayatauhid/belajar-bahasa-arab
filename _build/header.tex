% LaTeX preamble untuk paket Cahaya Tauhid
% Digunakan oleh pandoc → xelatex pipeline (lihat Makefile)

% Pandoc loads babel dengan bidi=basic (untuk Indonesian + RTL).
% Header ini menambah Arabic ke babel dan command \ayat dan \hadits.

% Tambah Arabic ke babel (pandoc cuma load Indonesian sebagai main)
\babelprovide[import]{arabic}

% Font Mushaf Madinah (Rasm Utsmani) untuk semua teks Arab dalam paket
\babelfont[arabic]{rm}[Script=Arabic, Scale=1.15]{Amiri Quran}

% Color palette dari logo Cahaya Tauhid
\usepackage{xcolor}
\definecolor{ctnavy}{HTML}{203543}
\definecolor{ctgold}{HTML}{D18C1C}

% Logo di title page
\usepackage{graphicx}
\usepackage{titling}
\pretitle{%

  \vspace*{0.5em}%
  \includegraphics[width=0.32\textwidth]{assets/logo/sumber/cahaya-tauhid.pdf}\par%
  \vspace{1.5em}%
  \LARGE\color{ctnavy}\bfseries%
}
\posttitle{\par\end{center}\vspace{1em}}

% Watermark logo semi-transparent di setiap halaman
% File watermark.png sudah di-pre-process dengan alpha 6%
% (lihat assets/logo/README.md untuk regenerasi)

\AddToShipoutPictureBG{%
  \AtPageCenter{%
    \makebox(0,0){%
      \includegraphics[width=0.55\paperwidth]{assets/logo/cahaya-tauhid-watermark.png}%
    }%
  }%
}

\usepackage{parskip}

% Spasi item dalam list (override pandoc \tightlist agar item yang wrap
% ke baris berikutnya jelas terbedakan dari item baru)
\AtBeginDocument{\renewcommand{\tightlist}{\setlength{\itemsep}{0.4em}\setlength{\parskip}{0pt}}}

% Ayat Al-Qur'an: rata tengah, RTL otomatis via babel arabic
\newcommand{\ayat}[1]{%
  \par\medskip%
  \begin{center}%
  {\Large\foreignlanguage{arabic}{#1}}%
  \end{center}%
  \medskip%
}

% Hadits / matan klasik
\newcommand{\hadits}[1]{%
  \par\medskip%
  \begin{center}%
  {\foreignlanguage{arabic}{#1}}%
  \end{center}%
  \medskip%
}

% Inline Arabic via \ar{...}
\newcommand{\ar}[1]{\foreignlanguage{arabic}{#1}}
